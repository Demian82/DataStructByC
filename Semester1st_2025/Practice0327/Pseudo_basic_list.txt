#include <stdio.h>

in main(void)
{
    1. 배열을 사용한 리스트의 초기화
    배열선언 및 초기화 data[]
    배열길이 변수 선언 length
    위치 변수 선언 pos
    삭제 값 지정 변수 선언 delItem
    인덱스 변수 선언 index

    3. 리스트의 마지막에 아이템 삽입
    data[length] = 아이템 값
    length++

    4. 리스트의 맨 앞에 아이템 삽입
    // 현재 항목들을 하나씩 뒤로 이동
    // 뒤에서부터 이동
    반복문
        //조건 (index = length-1; index >= 0; index--)
        data[index+1] = data[index]
    반복문 종료
    data[0] = 아이템
    length++


    5. 리스트의 중간에 아이템 삽입
    pos값 입력 // 실제 배열의 위치에 대응되는 값은 pos-1
    반복문
        for(index = length-1; index >= pos-1; index--)
        // 리스트의 마지막부터 pos위치까지의 데이터를
        // 한 칸씩 뒤로 미룸
        data[index+1] = data[index]
    반복문 종료
    data[pos-1] = 아이템
    length++

    6. 리스트의 맨 뒤의 아이템을 삭제
    delItem = data[length-1]
    length--
    // 배열의 길이는 배열의 선언 제한함
    // 배열의 데이터 검색에는 항상 length 변수를 사용
    // length 변수의 값이 제한되는 한,
    // 즉, length 변수의 범위가 제한되는 한,
    // 이 외의 범위는 검색 불가
    delItem 출력

    7. 리스트의 맨 앞의 아이템을 삭제
    delitem = data[0]
    두번째 데이터부터 마지막 데이터까지 한 칸씩 앞으로 당김
    length--
    delItem 출력

    8. 리스트의 중간의 아이템을 삭제
    pos값 입력
    delItem = data[pos-1]
    pos값 뒤에서부터의 데이터를 한 칸씩 앞으로 당김
    length--
    delItem 출력

    2. 리스트의 탐색 및 출력
    출력 (Linear list = [)
    반복문 for (index = 0; index < length; index++)
        출력 (index값 위치의 데이터)
        조건
            index가 마지막이 아니라면 ','를 붙여라
    반복문 끝
    출력 (])

}
